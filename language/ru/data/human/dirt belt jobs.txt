mission "Drought Relief"
	name `Облегчение засухи для <planet>`
	job
	repeat
	description `Сельское хозяйство <destination> в настоящее время переживает засуху. Доставить <cargo> на планету <date> для <payment>.`
	cargo "drought relief supplies" 25 2 .05
	deadline
	to offer
		random < 10
	source
		government "Republic"
		not attributes "farming"
	destination
		distance 3 10
		attributes "dirt belt"
		attributes "farming"
	on visit
		dialog phrase "общий груз в гостях"
	on complete
		payment 10000 160
		dialog "Вы сбрасываете свой груз <commodity> и получаете оплату <payment>. Группа граждан <planet> благодарит вас и прощается с вами."


mission "To Remembrance Day celebration [0]"
	name `Праздник Дня памяти`
	job
	repeat
	description `Принесите <fare> в <destination> от <date> для празднования Дня памяти 19 февраля. Платеж <payment>.`
	passengers 4 4 .5
	deadline
	to offer
		or
			and
				random < 20
				month == 12
				day >= 21
			and
				random < 30
				month == 1
			and
				random < 10
				month == 2
				day < 19
	to fail
		month * 100 + day >= 219
	source
		attributes "dirt belt"
	destination
		distance 3 15
		attributes "dirt belt"
	on visit
		dialog phrase "пассажир в гостях"
	on complete
		payment 1000 180
		dialog "Вы желаете своему <passengers> удачи на <planet>. Дети сбегают с вашего корабля в зеленой краске, а родители платят вам <payment>."
