mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Собрать 10 тонн алюминия"
	description "Добывайте и собирайте 10 тонн алюминия, а затем возвращайте минералы в Гегно на <origin>."
	source "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
	to accept
		"cargo space" > 10
	on visit
		dialog "Вы вернулись в <origin>, но в вашем грузовом отсеке нет 10 тонн алюминия. Не забудьте добыть 10 тонн алюминия и собрать его в свой груз, прежде чем приземлиться."
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno майнинг платежный диалог"
